function format(input){ let depth = 0; let lines = []; let cur = ""; for (let i = 0; i < input.length; i++) { let c = input[i]; if (c === '{' || c === '[') { if (cur.trim()) lines.push(" ".repeat(depth) + cur.trim()); lines.push(" ".repeat(depth) + c); depth++; cur = ""; } else if (c === '}' || c === ']') { if (cur.trim()) lines.push(" ".repeat(depth) + cur.trim()); depth--; lines.push(" ".repeat(depth) + c); cur = ""; } else if (c === ',') { cur += ','; lines.push(" ".repeat(depth) + cur.trim()); cur = ""; } else { cur += c; } } if (cur.trim()) lines.push(" ".repeat(depth) + cur.trim()); return lines; }
